-- Create a view for online users (active within last 30 minutes)
CREATE VIEW [dbo].[vw_OnlineUsers] AS
SELECT 
    u.Id,
    u.FirstName,
    u.LastName,
    u.Username,
    u.AvatarImg,
    us.Status,
    us.LastSeen
FROM 
    Users u
JOIN 
    UserStatus us ON u.Id = us.UserId
WHERE 
    us.LastSeen >= DATEADD(MINUTE, -30, GETUTCDATE());
GO

-- Create a stored procedure to update user status
CREATE PROCEDURE [dbo].[sp_UpdateUserStatus]
    @UserId UNIQUEIDENTIFIER,
    @Status NVARCHAR(20)
AS
BEGIN
    IF EXISTS (SELECT 1 FROM UserStatus WHERE UserId = @UserId)
    BEGIN
        UPDATE UserStatus 
        SET Status = @Status, 
            LastSeen = GETUTCDATE() 
        WHERE UserId = @UserId;
    END
    ELSE
    BEGIN
        INSERT INTO UserStatus (UserId, Status, LastSeen)
        VALUES (@UserId, @Status, GETUTCDATE());
    END
END;
